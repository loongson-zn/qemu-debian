105/** @defgroup ptr_safety Pointer Safety and Garbage Collection

107 *
108 * Utilities to assist with garbage collection in an implementation
109 * that supports <em>strict pointer safety</em>.
110 * This implementation only supports <em>relaxed pointer safety</em>
111 * and so these functions have no effect.
112 *
113 * C++11 20.6.4 [util.dynamic.safety], Pointer safety
114 *
115 * @{
116 */

118/// Constants representing the different types of pointer safety.
119enum class pointer_safety { relaxed, preferred, strict };
120
121/// Inform a garbage collector that an object is still in use.
122inline void
123declare_reachable(void*) { }
124
125/// Unregister an object previously registered with declare_reachable.
126template <typename _Tp>
127 inline _Tp*
128 undeclare_reachable(_Tp* __p) { return __p; }
129
130/// Inform a garbage collector that a region of memory need not be traced.
131inline void
132declare_no_pointers(char*, size_t) { }
133
134/// Unregister a range previously registered with declare_no_pointers.
135inline void
136undeclare_no_pointers(char*, size_t) { }
137
138/// The type of pointer safety supported by the implementation.
139inline pointer_safety
140get_pointer_safety() noexcept { return pointer_safety::relaxed; }

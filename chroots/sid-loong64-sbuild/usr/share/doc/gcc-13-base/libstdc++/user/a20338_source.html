72 // Returns __x ** __n, where __n >= 0. _Note that "multiplication"
73 // is required to be associative, but not necessarily commutative.
74 template<typename _Tp, typename _Integer, typename _MonoidOperation>
75 _Tp
76 __power(_Tp __x, _Integer __n, _MonoidOperation __monoid_op)
77 {
78 if (__n == 0)
79 return identity_element(__monoid_op);
80 else
81 {
82 while ((__n & 1) == 0)
83 {
84 __n >>= 1;
85 __x = __monoid_op(__x, __x);
86 }
87
88 _Tp __result = __x;
89 __n >>= 1;
90 while (__n != 0)
91 {
92 __x = __monoid_op(__x, __x);
93 if ((__n & 1) != 0)
94 __result = __monoid_op(__result, __x);
95 __n >>= 1;
96 }
97 return __result;
98 }
99 }

101 template<typename _Tp, typename _Integer>
102 inline _Tp
103 __power(_Tp __x, _Integer __n)
104 { return __power(__x, __n, std::multiplies<_Tp>()); }
